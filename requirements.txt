# ── Load logo SVG ──────────────────────────────────────────────────────────────
def _load_logo_b64():
    p = os.path.join(os.path.dirname(__file__), "assets", "novatech-logo.svg")
    if not os.path.exists(p):
        return "", ""
    with open(p) as f:
        raw = f.read()
    # White version (for dark bg)
    white = (raw
        .replace('fill="#1A1A1A"', 'fill="#F5F2EB"')
        .replace("fill='#1A1A1A'", "fill='#F5F2EB'")
        .replace('fill="#F9F9F7"/>', 'fill="#0A0A08"/>')
        .replace('rect width="400" height="400" fill="#F9F9F7"',
                 'rect width="400" height="400" fill="none"'))
    b64_white = base64.b64encode(white.encode()).decode()
    # Original (dark on light, for any light use)
    b64_orig  = base64.b64encode(raw.encode()).decode()
    return b64_white, b64_orig

LOGO_B64_WHITE, LOGO_B64_ORIG = _load_logo_b64()
